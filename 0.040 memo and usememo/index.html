1. Prevent unnecessary re-render

{`import React, { useState, memo } from 'react';

const Child = memo(({ name }) => {
  console.log('Child rendered');
  return <p>Hello, {name}!</p>;
});

function Parent() {
  const [count, setCount] = useState(0);
  return (
    <>
      <Child name="Pramish" />
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </>
  );
}`}

2. Cache expensive calculation

{`import React, { useState, useMemo } from 'react';

function Factorial({ number }) {
  const factorial = useMemo(() => {
    console.log('Calculating factorial');
    const calc = n => n <= 1 ? 1 : n * calc(n - 1);
    return calc(number);
  }, [number]);

  return <p>Factorial of {number} is {factorial}</p>;
}

function App() {
  const [number, setNumber] = useState(5);
  const [count, setCount] = useState(0);

  return (
    <>
      <Factorial number={number} />
      <input type="number" value={number} onChange={e => setNumber(+e.target.value)} />
      <p>Counter: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </>
  );
}`}

3. Multiple props memo

{`import React, { useState, memo } from 'react';

const Child = memo(({ firstName, lastName }) => {
  console.log('Child rendered');
  return <p>{firstName} {lastName}</p>;
});

function Parent() {
  const [firstName, setFirstName] = useState('Pramish');
  const [lastName, setLastName] = useState('Bhusal');
  const [count, setCount] = useState(0);

  return (
    <>
      <Child firstName={firstName} lastName={lastName} />
      <input value={firstName} onChange={e => setFirstName(e.target.value)} />
      <input value={lastName} onChange={e => setLastName(e.target.value)} />
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </>
  );
}`}

4. Filter large array efficiently

{`import React, { useState, useMemo } from 'react';

function FilterList({ numbers }) {
  const [query, setQuery] = useState('');
  const filtered = useMemo(() => numbers.filter(n => n.toString().includes(query)), [numbers, query]);

  return (
    <>
      <input value={query} onChange={e => setQuery(e.target.value)} placeholder="Filter numbers" />
      <ul>
        {filtered.map(n => <li key={n}>{n}</li>)}
      </ul>
    </>
  );
}`}

5. Combined memo & useMemo

{`import React, { useState, useMemo, memo } from 'react';

const Child = memo(({ text }) => {
  console.log('Child rendered');
  return <p>{text}</p>;
});

function App() {
  const [number, setNumber] = useState(5);
  const [text, setText] = useState('Hello');

  const factorial = useMemo(() => {
    const calc = n => n <=1 ? 1 : n * calc(n-1);
    return calc(number);
  }, [number]);

  return (
    <>
      <p>Factorial of {number}: {factorial}</p>
      <input value={number} onChange={e => setNumber(+e.target.value)} />
      <Child text={text} />
      <input value={text} onChange={e => setText(e.target.value)} />
    </>
  );
}`}
