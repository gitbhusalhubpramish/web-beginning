2. Input Mirror

import { useState } from 'react';

function InputMirror() {
  const [text, setText] = useState('');
  return (
    <>
      <input value={text} onChange={e => setText(e.target.value)} placeholder="Type here" />
      <p>{text}</p>
    </>
  );
}

3. Form Alert

import { useState } from 'react';

function FormAlert() {
  const [text, setText] = useState('');
  const handleSubmit = e => {
    e.preventDefault();
    alert('Submitted: ' + text);
  };
  return (
    <form onSubmit={handleSubmit}>
      <input value={text} onChange={e => setText(e.target.value)} placeholder="Enter text" />
      <button type="submit">Submit</button>
    </form>
  );
}

4. Key Press Logger

function KeyLogger() {
  const handleKeyDown = e => console.log('Key pressed:', e.key);
  return <input onKeyDown={handleKeyDown} placeholder="Press a key" />;
}

5. Focus & Blur Messages

import { useState } from 'react';

function FocusBlur() {
  const [msg, setMsg] = useState('');
  return (
    <>
      <input
        onFocus={() => setMsg('Input focused')}
        onBlur={() => setMsg('Input blurred')}
        placeholder="Focus me"
      />
      <p>{msg}</p>
    </>
  );
}

6. Mouse Hover Color Change

import { useState } from 'react';

function HoverBox() {
  const [bg, setBg] = useState('white');
  return (
    <div
      className="hoverBox"
      style={{ backgroundColor: bg }}
      onMouseEnter={() => setBg('lightblue')}
      onMouseLeave={() => setBg('white')}
    >Hover me</div>
  );
}

9. Drag & Drop

{`import { useState } from 'react';

function DragDrop() {
  const [dropped, setDropped] = useState(false);

  const handleDrop = e => {
    e.preventDefault();
    setDropped(true);
  };

  const handleDragOver = e => e.preventDefault();

  return (
    <>
      <div draggable style={{ width: '100px', height: '50px', backgroundColor: 'lightgreen' }}>Drag me</div>
      <div className="dropTarget" onDrop={handleDrop} onDragOver={handleDragOver}>
        {dropped ? 'Dropped!' : 'Drop here'}
      </div>
    </>
  );
}`}
